def pretty_print(pp)
    pp.text "cannot use #{selected_model.short_name} for the child #{child_name} of #{composition_model.short_name}"
    pp.breakable
    pp.text "it does not provide the required models"
    pp.nest(2) do
        pp.breakable
        pp.seplist(required_models) do |m|
            pp.text m.short_name
        end
    end
end
