def require_dynamic_service(dynamic_service_name, as: nil, **dyn_options)
    @child = @context.specialized_child(@child)
    srv = @child.model.require_dynamic_service(dynamic_service_name, as: as, **dyn_options)
    srv.attach(@child)
end
