def proxy(peer)
    component_model = peer.local_object(self.component_model)
    if srv = component_model.find_data_service(name)
        return srv
    else
        BoundDataService.new(name, component_model, peer.local_object(master), peer.local_object(model), Hash.new)
    end
end
