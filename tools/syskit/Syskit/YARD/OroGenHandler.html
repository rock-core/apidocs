def parse_superclass(statement)
    # We assume that all classes in OroGen have Syskit::TaskContext
    # as superclass by default
    if !statement
        statement = ::YARD.parse_string("Syskit::TaskContext").
            enumerator.first
    end
    super(statement)
end
