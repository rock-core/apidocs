def pretty_print(pp)
    if !instanciation_chain.empty?
        pp.text "while instanciating"
        pp.nest(2) do
            pp.breakable
            pp.seplist(instanciation_chain.reverse) do |m|
                m.pretty_print(pp)
            end
        end
    end
end
