def add_conf_item(label, accessor = nil)
    item1 = Vizkit::VizkitItem.new(label)
    item2 = RubyItem.new

    if !accessor.nil?
        item2.getter do
            @editing_model.method(accessor).call
        end

        item2.setter do |value|
            @editing_model.method("#{accessor}=".to_sym).call value
        end
    end

    appendRow([item1, item2])
    return item1, item2
end
