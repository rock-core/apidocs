def each_submodel(model)
    if model == Syskit::TaskContext
        model.each_submodel do |m|
            excluded = (!m.name || m.private_specialization?)
            yield(m, excluded)
        end
    end
end

def split_name(model)
    name = model.name
    if name.start_with?("OroGen.")
        name.split(".")
    else
        name.split("::")
    end
end
