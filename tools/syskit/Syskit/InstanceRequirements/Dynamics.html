def merge(other)
    task.merge(other.task)
    other.ports.each_key { |port_name| ports[port_name] ||= NetworkGeneration::PortDynamics.new(port_name) }
    @ports = ports.merge(other.ports) do |port_name, old, new|
        old.merge(new)
    end
end
