def droby_dump(peer)
    peer_registry = peer.object_manager.typelib_registry
    if !peer_registry.include?(name)
        reg = registry.minimal(name)
        xml = reg.to_xml
        peer_registry.merge(reg)
    end
    DRoby.new(name, xml)
end
