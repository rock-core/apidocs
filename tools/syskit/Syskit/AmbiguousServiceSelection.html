def pretty_print(pp)
    pp.text "there is an ambiguity while looking for a service of type #{required_service} in #{task_model.short_name}"
    pp.breakable
    pp.text "candidates are:"
    pp.nest(2) do
        pp.breakable
        pp.seplist(candidates) do |service|
            pp.text(service.name || service.to_s)
        end
    end
end
