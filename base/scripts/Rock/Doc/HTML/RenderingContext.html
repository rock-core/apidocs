def link_to(arg)
    text =
        if arg.respond_to?(:name)
            arg.name
        else arg.to_s
        end

    Doc::HTML.escape_html(text)
end

def render_item(name, value = nil)
    if value
        "<li><b>#{name}</b>: #{value}</li>"
    else
        "<li><b>#{name}</b></li>"
    end
end
